## append():- Adds an element to the end of the list.
# example:
me = ['food','sleep']
me.append('read')
print(me) # this will print "['food','sleep','read']"

## clear():-Removes all the elements from the list
# example:
body = ['head','nose','eye']
body.clear()
print(body) # this will print "[]"

## insert():-  Inserts an element at a specified index.
# example:
greatings = ['hello','ola']
greatings.insert(0,'selam') # this will print "['selam','hello','ola']"
